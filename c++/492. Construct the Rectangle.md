```c++
class Solution {
public:
    vector<int> constructRectangle(int area) {
        vector<int>ans(2,0);
        int Min = INT_MAX;
        for(int i = 1;i<=sqrt(area);i++)
        {
            if(area%i == 0 && area%i - i <Min)
            {
                ans[0] = area/i;
                ans[1] = i;
                Min = ans[0]-ans[1];
            }
        }
        return ans;
    }
};
```
