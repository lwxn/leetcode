```c++
/*
// Definition for a Node.
class Node {
public:
    int val;
    vector<Node*> children;

    Node() {}

    Node(int _val, vector<Node*> _children) {
        val = _val;
        children = _children;
    }
};
*/
class Solution {
private:
    vector<int>ans;
public:
    vector<int> postorder(Node* root) {
        print(root);
        return ans;
    }
    
    void print(Node* root)
    {
        if(root)
        {
            for(Node* child : root->children) print(child);
            ans.push_back(root->val);
            
        }
    }
};
```
