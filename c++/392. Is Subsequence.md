```c++
class Solution {
public:
    bool isSubsequence(string s, string t) {
        if(s.size()==0) return 1;
        int j = 0;
        for(int i = 0;i<t.size();i++)
        {
            if(t[i] == s[j])
            {
                j++;
                cout<<j<<endl;
            }
            if(j == s.size()) return true;
        }
        return false;
    }
};
```
